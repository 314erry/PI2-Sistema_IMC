def calculo_matematico():
    x = 0
    while True:
        x += math.sqrt(x)

b) faça um gerador de arquivos de 10 Mb, contendo strings aleatórias que rode em loop.
# Programa para gerar arquivos de 10 MB com strings aleatórias
import string
import random

def gerar_arquivo():
    while True:
        with open('arquivo.txt', 'w') as f:
            f.write(''.join(random.choices(string.ascii_letters + string.digits, k=10485760))) # 10 MB

c) faça um monitor de processos de encerre processos que:
   I) estejam consumindo mais de 30% de CPU
  II) que finalize o gerador de arquivos aleatórios quando o consumo do disco ultrapassar 60%
  II) que finalize programas rodando a mais de 2 minutos

# Programa para monitorar e finalizar processos de acordo com as regras especificadas
import psutil
import subprocess
import time

def monitorar_processos():
    while True:
        for proc in psutil.process_iter():
            try:
                # Regra I: Finaliza processos que estão consumindo mais de 30% de CPU
                if proc.cpu_percent() > 30:
                    proc.kill()
                
                # Regra II: Finaliza o gerador de arquivos aleatórios quando o consumo do disco ultrapassar 60%
                if psutil.disk_usage('/').percent > 60 and 'gerar_arquivo.py' in proc.cmdline():
                    proc.kill()
                
                # Regra III: Finaliza programas rodando a mais de 2 minutos
                if time.time() - proc.create_time() > 120:
                    proc.kill()
            except (psutil.NoSuchProcess, psutil.AccessDenied, psutil.ZombieProcess):
                pass

monitorar_processos()
